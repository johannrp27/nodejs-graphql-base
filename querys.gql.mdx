# Mutations

``` graphql
  mutation($productInput: CreateProductInput!) {
    createProduct(productInput: $productInput ) {
      id
      name,
      createdAt
    }
  }

  mutation(
    $updateProductId: ID!,
    $updateProductProductInput2: UpdateProductInput!
  ) {
    updateProduct(
      id: $updateProductId,
      productInput: $updateProductProductInput2
    ) {
      id
      name
      image
      createdAt
    }
  }

  mutation(
    $deleteProductId: ID!
  ) {
    deleteProduct(id: $deleteProductId)
  }
```
